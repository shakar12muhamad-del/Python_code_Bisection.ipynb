        "import numpy as np\n",
        "import matplotlib.pyplot as plt\n",
        "\n",
        "\n",
        "def get_function():\n",
        "    expr = input(\"Enter the function f(x): (e.g. x**3 - 4*x - 9)\\n>>> \")\n",
        "    def f(x):\n",
        "        return eval(expr, {\"x\": x, \"np\": np})\n",
        "    return f, expr\n",
        "\n",
        "\n",
        "def bisection(f, a, b, tol=1e-6, max_iter=100):\n",
        "    if f(a) * f(b) >= 0:\n",
        "        print(\"Error: f(a) and f(b) must have opposite signs.\")\n",
        "        return None\n",
        "\n",
        "    print(f\"\\n{'Iter':<6}{'a':<12}{'b':<12}{'c':<12}{'f(c)':<12}{'Error':<12}\")\n",
        "    print(\"-\" * 60)\n",
        "\n",
        "    c_old = a\n",
        "    for i in range(max_iter):\n",
        "        c = (a + b) / 2\n",
        "        fc = f(c)\n",
        "        error = abs(c - c_old)\n",
        "        print(f\"{i+1:<6}{a:<12.6f}{b:<12.6f}{c:<12.6f}{fc:<12.6f}{error:<12.6f}\")\n",
        "\n",
        "        if abs(fc) < tol or error < tol:\n",
        "            print(\"\\n✅ Root found at x =\", c)\n",
        "            return c\n",
        "\n",
        "        if f(a) * fc < 0:\n",
        "            b = c\n",
        "        else:\n",
        "            a = c\n",
        "\n",
        "        c_old = c\n",
        "\n",
        "    print(\"\\n⚠️ Method did not converge within the given iterations.\")\n",
        "    return None\n",
        "\n",
        "\n",
        "\n",
        "f, expr = get_function()\n",
        "a = float(input(\"Enter the lower limit a: \"))\n",
        "b = float(input(\"Enter the upper limit b: \"))\n",
        "tol = float(input(\"Enter Error (1e-6): \"))\n",
        "max_iter = int(input(\"Enter maximum number of iterations: \"))\n",
        "\n",
        "\n",
        "root = bisection(f, a, b, tol, max_iter)\n",
        "\n",
        "\n",
        "if root is not None:\n",
        "    x = np.linspace(a - 1, b + 1, 400)\n",
        "    y = [f(i) for i in x]\n",
        "\n",

        "    plt.axvline(root, color='red', linestyle='--', label=f'Root ≈ {root:.6f}')\n",
        "    plt.title(\"Bisection Method Visualization\")\n",
        "    plt.xlabel(\"x\")\n",
        "    plt.ylabel(\"f(x)\")\n",
        "    plt.legend()\n",
